\begin{homeworkProblem}
Give expression for $t$ and $y$:
\begin{align*}
  t &= w(wu_1 + u_2) + u_3 \\
  y &= w(w(wu_1 + u_2) + u_3)
\end{align*}
Compute $\frac{\d y}{\d w}$ and $\frac{\p y}{\p p}$:
\begin{align*}
  \frac{\d y}{\d w} &= 3w^2 u_1 + 2wu_2 + u_3 \\
  \frac{\partial y}{\partial p} &= w^3
\end{align*}

\end{homeworkProblem}
